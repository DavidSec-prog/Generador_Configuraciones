// ===== Validaciones =====
bool validarIP(const string& ip) {
    stringstream ss(ip);
    string segmento;
    int count = 0;
    while (getline(ss, segmento, '.')) {
        if (segmento.empty()) return false;
        try {
            int valor = stoi(segmento);
            if (valor < 0 || valor > 255) return false;
        } catch (...) { return false; }
        count++;
    }
    return (count == 4);
}

bool validarVLAN(const string& vlan) {
    try {
        int valor = stoi(vlan);
        return (valor >= 2 && valor <= 4094);
    } catch (...) { return false; }
}

// ===== Calculadora IP =====
vector<int> ipToVector(const string& ip) {
    vector<int> octetos;
    stringstream ss(ip);
    string segmento;
    while (getline(ss, segmento, '.')) octetos.push_back(stoi(segmento));
    return octetos;
}

string vectorToIp(const vector<int>& o) {
    return to_string(o[0]) + "." + to_string(o[1]) + "." +
           to_string(o[2]) + "." + to_string(o[3]);
}

vector<int> prefixToMask(int p) {
    vector<int> m(4,0);
    for (int i=0; i<4; i++) {
        if (p >= 8) { m[i] = 255; p -= 8; }
        else if (p > 0) { m[i] = (256 - (1 << (8-p))); p = 0; }
    }
    return m;
}

vector<int> maskToWildcard(const vector<int>& m) {
    vector<int> w(4);
    for (int i=0; i<4; i++) w[i] = 255 - m[i];
    return w;
}

vector<int> calcularNetwork(const vector<int>& ip, const vector<int>& m) {
    vector<int> n(4);
    for (int i=0; i<4; i++) n[i] = ip[i] & m[i];
    return n;
}

vector<int> calcularBroadcast(const vector<int>& ip, const vector<int>& m) {
    vector<int> b(4);
    for (int i=0; i<4; i++) b[i] = ip[i] | (255 - m[i]);
    return b;
}

vector<int> sumarIp(const vector<int>& ip, int delta) {
    unsigned int val = (ip[0]<<24) | (ip[1]<<16) | (ip[2]<<8) | ip[3];
    val += delta;
    vector<int> r(4);
    r[0] = (val>>24) & 255;
    r[1] = (val>>16) & 255;
    r[2] = (val>>8) & 255;
    r[3] = val & 255;
    return r;
}

// ===== Plantilla Huawei AR617 - Internet =====
void configurarHuaweiAR617Internet() {
    string nombreCliente, codigoServicio;
    string ipWAN, mascaraWAN, ipLAN, mascaraLAN, vlan;
    string ipPrivada, mascaraPrivada;
    char opcionDHCP;

    cout << "=== Configuracion de Router Huawei 617 ===" << endl;

    cout << "Nombre del cliente (MAYUSCULA): "; getline(cin, nombreCliente);
    cout << "Codigo de servicio(MAYUSCULA): "; getline(cin, codigoServicio);

    // WAN
    do {
        cout << "Direccion IP WAN: ";
        getline(cin, ipWAN);
    } while (!validarIP(ipWAN));

    int prefijoWAN = 0;
    while (true) {
        cout << "Mascara WAN (aceptado: 21-24 o 30): ";
        getline(cin, mascaraWAN);
        try { prefijoWAN = stoi(mascaraWAN); } catch (...) { prefijoWAN = 0; }
        if (prefijoWAN == 30 || (prefijoWAN >= 21 && prefijoWAN <= 24)) break;
        else cout << "Dato invalido. Ingrese nuevamente." << endl;
    }

    do {
        cout << "VLAN (2-4094): ";
        getline(cin, vlan);
    } while (!validarVLAN(vlan));

    // LAN pública
    do {
        cout << "Direccion IP LAN publica: ";
        getline(cin, ipLAN);
    } while (!validarIP(ipLAN));

    cout << "Mascara LAN (ej: 29 o 30): ";
    getline(cin, mascaraLAN);
    int prefijoLAN = stoi(mascaraLAN);

    cout << "Desea habilitar DHCP? (s/n): ";
    cin >> opcionDHCP;
    cin.ignore();

    if (opcionDHCP=='s' || opcionDHCP=='S') {
        do {
            cout << "Direccionamiento privado (ej: 192.168.1.0): ";
            getline(cin, ipPrivada);
        } while (!validarIP(ipPrivada));
        cout << "Mascara direccionamiento privado (por defecto 24): ";
        getline(cin, mascaraPrivada);
        if (mascaraPrivada.empty()) mascaraPrivada = "24";
    }

    // ===== Calculadora WAN =====
    vector<int> ipVecWAN = ipToVector(ipWAN);
    vector<int> maskWAN = prefixToMask(prefijoWAN);
    vector<int> wildcardWAN = maskToWildcard(maskWAN);
    vector<int> networkWAN = calcularNetwork(ipVecWAN, maskWAN);
    vector<int> broadcastWAN = calcularBroadcast(ipVecWAN, maskWAN);
    vector<int> hostMinWAN = sumarIp(networkWAN, 1);
    vector<int> hostMaxWAN = sumarIp(broadcastWAN, -1);
    string ipParaConfigWAN = (prefijoWAN == 30) ? vectorToIp(hostMaxWAN) : ipWAN;

    // ===== Calculadora LAN =====
    vector<int> ipVecLAN = ipToVector(ipLAN);
    vector<int> maskLAN = prefixToMask(prefijoLAN);
    vector<int> networkLAN = calcularNetwork(ipVecLAN, maskLAN);
    vector<int> hostMinLAN = sumarIp(networkLAN, 1);

    // ===== Calculadora Privada (si aplica) =====
    vector<int> ipVecPriv, maskPriv, networkPriv, hostMinPriv, wildcardPriv;
    int prefijoPriv = 0;
    if (opcionDHCP=='s' || opcionDHCP=='S') {
        prefijoPriv = stoi(mascaraPrivada);
        ipVecPriv = ipToVector(ipPrivada);
        maskPriv = prefixToMask(prefijoPriv);
        networkPriv = calcularNetwork(ipVecPriv, maskPriv);
        hostMinPriv = sumarIp(networkPriv, 1);
        wildcardPriv = maskToWildcard(maskPriv);
    }

    // ===== Plantilla de Configuración =====
    string prefijo = nombreCliente.substr(0, 4);
    string comandosysname = "sysname " + prefijo + "_" + codigoServicio;

   cout << "\033[33m=== COMANDOS DE CONFIGURACION HUAWEI 617 ===\033[0m" << endl;
    cout << comandosysname << endl;

    cout << "" << endl;
    // WAN
    cout << "interface GigabitEthernet0/0/4." << vlan << endl;
    cout << " description --WAN INTERNET --" << codigoServicio << "--" << endl;
    cout << " dot1q termination vid " << vlan << endl;
cout << " ip address " << ipParaConfigWAN << " " << vectorToIp(maskWAN) << endl;
 cout << "quit" << endl;
 cout << "" << endl;
    cout << "ip route-static 0.0.0.0 0.0.0.0 GigabitEthernet0/0/4." << vlan
         << " " << vectorToIp(hostMinWAN) << endl;
    cout << "" << endl;
    cout << "acl number 2300" << endl;
cout << " rule 1 permit source " << vectorToIp(networkWAN) << " " << vectorToIp(wildcardWAN) << endl;
 cout << "" << endl;
    cout << "interface GigabitEthernet0/0/4" << endl;
    cout << " combo-port copper\n undo negotiation auto\n speed 1000\n duplex full\n quit\n" << endl;
    cout << "" << endl;
    

    // LAN
    cout << "interface Vlanif1" << endl;
    cout << " description --LAN INTERNET --" << codigoServicio << endl;
    cout << " ip address " << vectorToIp(hostMinLAN) << " " << vectorToIp(maskLAN) << endl;
     
    if (opcionDHCP=='s' || opcionDHCP=='S') {
        cout << " ip address " << vectorToIp(hostMinPriv) << " " << vectorToIp(maskPriv) << " sub" << endl;
        cout << " dhcp select global" << endl;
        cout << "quit" << endl;
        cout << "" << endl;
        cout << "nat address-group 1 " << vectorToIp(networkLAN) << " " << vectorToIp(networkLAN) << endl;
        cout << "" << endl;
	cout << "acl number 2000" << endl;
        cout << " rule 1 permit source " << vectorToIp(networkPriv) << " " << vectorToIp(wildcardPriv) << endl;
        cout << "quit" << endl;        
        cout << "" << endl;
        cout << "interface GigabitEthernet0/0/4." << vlan << endl;
        cout << " nat outbound 2000 address-group 1" << endl;
        cout << "quit" << endl;
        cout << "" << endl;
        cout << "ip pool DHCP-LAN_privada" << endl;
        cout << " gateway-list " << vectorToIp(hostMinPriv) << endl;
        cout << " network " << vectorToIp(networkPriv) << " mask " << vectorToIp(maskPriv) << endl;
        cout << " dns-list 190.157.8.33 200.14.207.210" << endl;
    }

    cout << "\033[33m=================================\033[0m" << endl;
}
